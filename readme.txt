```

│   └── main.cpp          # 主程序入口

├── docs/                  # 项目文档
├── config/                # 配置文件目录
├── build/                 # 构建输出目录
├── scripts/               # 构建脚本
├── CMakeLists.txt         # 主构建文件
└── README.md              # 项目说明
```
